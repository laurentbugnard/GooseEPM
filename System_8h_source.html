75template <class T>
76inline T create_distance_lookup(const T& distance)
77{
78 using value_type = typename T::value_type;
79 static_assert(std::numeric_limits<value_type>::is_integer, "Distances must be integer");
80 static_assert(std::numeric_limits<value_type>::is_signed, "Distances must be signed");
81 GOOSEEPM_REQUIRE(distance.dimension() == 1, std::invalid_argument);
82
83 value_type lower = detail::amin(distance);
84 value_type upper = detail::amax(distance) + 1;
85 auto d = xt::arange<value_type>(lower, upper);
86 GOOSEEPM_REQUIRE(xt::all(xt::in1d(distance, d)), std::invalid_argument);
87 GOOSEEPM_REQUIRE(distance.size() == upper - lower, std::invalid_argument);
88
89 value_type N = static_cast<value_type>(distance.size());
90 T ret = xt::empty<value_type>({2 * N - 1});
91
92 for (value_type i = 0; i < upper; ++i) {
93 ret(i) = detail::argmax(xt::equal(distance, i));
94 }
95 for (value_type i = upper; i < N; ++i) {
96 ret(i) = detail::argmax(xt::equal(distance, i - N));
97 }
98 for (value_type i = -1; i >= lower; --i) {
99 ret.periodic(i) = detail::argmax(xt::equal(distance, i));
100 }
101 for (value_type i = lower; i > -N; --i) {
102 ret.periodic(i) = detail::argmax(xt::equal(distance, N + i));
103 }
104
105 return ret;
106}

328 void initSigmaFast(double sigma_std, size_t delta_r)
329 {
330 m_sig.fill(0);
331 ptrdiff_t d = static_cast<ptrdiff_t>(delta_r);
332
333 for (ptrdiff_t i = 0; i < m_sig.shape(0); ++i) {
334 for (ptrdiff_t j = 0; j < m_sig.shape(1); ++j) {
335 double dsig = m_gen.normal(std::array<size_t, 1>{1}, 0, sigma_std)(0);
336 m_sig(i, j) += dsig;
337 m_sig.periodic(i - d, j) -= 0.5 * dsig;
338 m_sig.periodic(i + d, j) -= 0.5 * dsig;
339 m_sig.periodic(i, j - d) -= 0.5 * dsig;
340 m_sig.periodic(i, j + d) -= 0.5 * dsig;
341 m_sig.periodic(i + d, j + d) += 0.25 * dsig;
342 m_sig.periodic(i - d, j + d) += 0.25 * dsig;
343 m_sig.periodic(i + d, j - d) += 0.25 * dsig;
344 m_sig.periodic(i - d, j - d) += 0.25 * dsig;
345 }
346 }
347 m_sig *= 0.5;
348 }

398 size_t makeFailureStep()
399 {
400 auto failing = xt::argwhere(m_sig < -m_sigy || m_sig > m_sigy);
401 size_t nfailing = failing.size();
402 m_t += m_gen.exponential(std::array<size_t, 1>{1}, m_failure_rate * nfailing)(0);
403 size_t i = m_gen.randint(std::array<size_t, 1>{1}, static_cast<size_t>(nfailing - 1))(0);
404 size_t idx = m_sig.shape(0) * failing[i][0] + failing[i][1];
405 this->spatialParticleFailure(idx);
406 return idx;
407 }

437 void spatialParticleFailure(size_t idx)
438 {
439 double dsig = m_sig.flat(idx) + m_gen.normal(std::array<size_t, 1>{1}, 0.0, 0.01)(0);
440
441 m_sig.flat(idx) -= dsig;
442 m_epsp.flat(idx) += dsig;
443 m_sigy.flat(idx) =
444 m_gen.normal(std::array<size_t, 1>{1}, m_sigy_mu.flat(idx), m_sigy_std.flat(idx))(0);
445
446 auto index = xt::unravel_index(idx, m_sig.shape());
447 ptrdiff_t i0 = static_cast<ptrdiff_t>(index[0]);
448 ptrdiff_t j0 = static_cast<ptrdiff_t>(index[1]);
449
450 for (ptrdiff_t i = 0; i < m_sig.shape(0); ++i) {
451 for (ptrdiff_t j = 0; j < m_sig.shape(1); ++j) {
452 if (i == i0 && j == j0) {
453 continue;
454 }
455 m_sig(i, j) +=
456 m_propagator(m_drow.periodic(i - i0), m_drow.periodic(j - j0)) * dsig;
457 }
458 }
459
460 if (!m_fixed_stress) {
461 m_sigbar -= dsig / static_cast<double>(m_sig.size());
462 }
463
464 m_sig -= detail::mean(m_sig) - m_sigbar;
465 }
